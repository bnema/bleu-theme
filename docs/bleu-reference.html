Syntax Highlighting Example

// Example TypeScript code with Bleu theme syntax highlighting
import { Component } from 'react';

interface UserProps {
  name: string;
  age: number;
  isActive: boolean;
}

const greetUser = (user: UserProps): string => {
  if (user.isActive) {
    return `Hello, ${user.name}! You are ${user.age} years old.`;
  }
  return 'User is not active';
};

// Usage
const user = {
  name: 'Alice',
  age: 30,
  isActive: true
};

console.log(greetUser(user));

Accessibility & Contrast Ratios

WCAG Compliance

All text colors meet WCAG accessibility standards:

  • Primary Text (#e0ecf4) on Deep Navy: ~13:1 (AAA)
  • Pure White (#fefefe) on Deep Navy: ~16:1 (AAA)
  • Dimmed Text (#708090) on Deep Navy: ~5:1 (AA)
  • Bright Cyan (#00d4ff) on Deep Navy: ~11:1 (AAA)
